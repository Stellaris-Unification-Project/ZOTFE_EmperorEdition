country_event = {
	id = fallen_weaponry.100
	title = fallen_weaponry.100.name
	desc = fallen_weaponry.100.desc
	picture = GFX_evt_underground_city
	location = from
	show_sound = event_mystic_reveal
	fire_only_once = yes
	trigger = {
		host_has_dlc = "Ancient Relics Story Pack"
		has_global_flag = zofe_secrets_of_the_fallen_empires_activated
		is_country_type = ascended_empire
		has_technology = tech_fallen_technologies_4
		has_technology = tech_fallen_technologies_6
		any_owned_planet = { has_planet_flag = fallen_spiritualist_capital }
		NOT = { has_country_flag = fallen_spiritualist_capital_raided }
	}
	mean_time_to_happen = { months = 5 }
	immediate = {
		set_country_flag = fallen_spiritualist_capital_raided
		random_owned_planet = {
			limit = { has_planet_flag = fallen_spiritualist_capital }
			save_event_target_as = fallen_spiritualist_digsite
		}
	}
	option = {
		name = CURIOUS
		event_target:fallen_spiritualist_digsite = { create_archaeological_site = fallen_spiritualist_digsite_1 }
		owner = {
			if = {
				limit = { NOT = { has_event_chain = lost_technologies_chain } }
				begin_event_chain = {
					event_chain = "lost_technologies_chain"
					target = this
				}
			}
		}
		owner = {
			add_monthly_resource_mult = {
				resource = engineering_research
     			value = @tier1researchreward
     			min = @tier1researchmin
     			max = @tier1researchmax
     		}
		}
	}
}
